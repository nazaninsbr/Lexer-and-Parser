		(VariableDeclaration)* (MainStatement)* ReturnExpression
	;

MainStatement
	:
		AtomicMainStatement
		| Statement
	;

		['a'..'z' | 'A'..'Z' | '0'..'9' | COLON | '%' | LParentheses | RParentheses | LBrackets | RBrackets | LSquareBrackets | RSquareBrackets | '#' | Delimiter | '&' | '^' | BinaryArithmaticalOperatorsPriorityTwo | '@' | COMMA | BinaryLogicalOperators | UnaryLogicalOperators | Dollor | BinaryArithmaticalOperatorsPriorityOne | Underscore | AssignmentOperator | ComparisonOperators | '~' | Quotation]*
	;

MainPrintStatement
	:
		PrintStatement
		| PrintCommand LParentheses Expression RParentheses
	;

AtomicMainStatement
	:
		( ClassInstantiation
		| Identifier AssignmentOperator ClassInstantiation
		| MainPrintStatement
		) Delimeter

	;

MainExpression

		MainMatheticalExpression | MainLogicalExpression | String | BooleanValue
	;

		MainMultiplyExpression

	;

	:
		MainAtomExpression
        ( BinaryArithmaticalOperatorsPriorityOne MainAtomExpression )*
    ;

		MathematicalTerm
		| (UnaryArithmaticalOperators |  ) LParentheses MainMatheticalExpression RParentheses 
		| MainLogicalExpression
	;

		(UnaryArithmaticalOperators |  ) MainNumericalTerm
	;

		| Number
		| ArrayAccess
		| SelfVariableAccess
		| SelfMethodAccess
		| ArrayLength
		| ClassInstantiation
	;

MainLogicalExpression
	:
		UnaryLogicalOperators MainLogicalAtomExpression MainHalfLogicalExpression MainLogicalExpressionPrime 
		| LParentheses LogicalExpression RParentheses MainLogicalExpressionPrime
		| MainMatheticalExpression (ComparisonOperators | BinaryLogicalOperators) MainLogicalAtomExpression MainHalfLogicalExpression MainLogicalExpressionPrime
		| MainLogicalTerm (ComparisonOperators | BinaryLogicalOperators) MainLogicalAtomExpression MainHalfLogicalExpression MainLogicalExpressionPrime
	;

MainHalfLogicalExpression 
	:
		((ComparisonOperators | BinaryLogicalOperators) MainLogicalAtomExpression)*
	;

MainLogicalExpressionPrime
	:
		((ComparisonOperators | BinaryLogicalOperators) MainLogicalAtomExpression MainHalfLogicalExpression MainLogicalExpressionPrime
		|  
	;

MainLogicalAtomExpression

		| MainLogicalTerm
		| MatheticalExpression
	;

		Identifier
		| Number

		| ArrayLength
		| ClassInstantiation
	;
